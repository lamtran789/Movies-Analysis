// Had to group ratings_table --> links_table <-- movies_metadata. Also decided to keep movie per row with the averages out of 5 for each movie. Limit to 100 so queries run smoother and main data still represented.

with group_ratings as (
select moveid, avg(rating) as avg_rating
from ratings
group by moveid)
select l.movieid, l.imbdid, l.tmdbld, gr.moveid, gr.avg_rating, m.budget, m.id, m.imdb_id, m.popularity, m.release, m.revenue, m.runtime,
m.title, m.vote_avg, m.vote_count
from links l
inner join group_ratings gr
on gr.moveid=l.movieid
inner join movies m
on l.imbdid=m.imdb_id
limit 100

// Placed this entire table into another CTE to make it easier to analyze the data. Also unselected some ID columns to make the dataset cleaner and more focused on results. Also formatted the query to make it look cleaner and easier to read.

WITH
	FINAL_DATA AS (
		WITH
			GROUP_RATINGS AS (
				SELECT
					MOVEID,
					AVG(RATING) AS AVG_RATING
				FROM
					RATINGS
				GROUP BY
					MOVEID
			)
		SELECT
			L.MOVIEID,
			GR.AVG_RATING,
			M.BUDGET,
			M.POPULARITY,
			M.RELEASE,
			M.REVENUE,
			M.RUNTIME,
			M.TITLE,
			M.VOTE_AVG,
			M.VOTE_COUNT
		FROM
			LINKS L
			INNER JOIN GROUP_RATINGS GR ON GR.MOVEID = L.MOVIEID
			INNER JOIN MOVIES M ON L.IMBDID = M.IMDB_ID
		LIMIT
			100
	)
SELECT
	*
FROM
	FINAL_DATA
